// In your firestore.rules file, add this rule for the counters collection:

match /counters/{document} {
  // Allow authenticated users to read and update the order counter
  allow read, write: if request.auth != null && document == 'order_counter';
}

// Alternative: If you want to be more restrictive, only allow updates:
match /counters/order_counter {
  // Allow authenticated users to update the order counter (for atomic increments)
  allow read, update: if request.auth != null;
}
